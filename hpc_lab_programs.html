#include <bits/stdc++.h>
#include <omp.h>
using namespace std;

const int NODES = 7;
vector<int> adj[NODES];
bool visited[NODES];
omp_lock_t lk;

void parallel_bfs(int src) {
    fill(visited, visited+NODES, false);
    queue<int> q;
    visited[src] = true;
    q.push(src);
    cout << "BFS: ";
    while (!q.empty()) {
        int sz = q.size();
        vector<int> lvl(sz);
        for (int i = 0; i < sz; i++) { lvl[i] = q.front(); q.pop(); }

        #pragma omp parallel for
        for (int i = 0; i < sz; i++) {
            cout << lvl[i] << " ";
            for (int nb : adj[lvl[i]]) {
                omp_set_lock(&lk);
                if (!visited[nb]) { visited[nb]=true; q.push(nb); }
                omp_unset_lock(&lk);
            }
        }
    }
    cout << "\n";
}

void dfs_util(int u) {
    visited[u] = true;
    cout << u << " ";
    #pragma omp parallel for
    for (int i = 0; i < (int)adj[u].size(); i++) {
        int v = adj[u][i];
        bool go = false;
        #pragma omp critical
        if (!visited[v]) { visited[v]=true; go=true; }
        if (go) dfs_util(v);
    }
}

void parallel_dfs(int src) {
    fill(visited, visited+NODES, false);
    cout << "DFS: ";
    dfs_util(src);
    cout << "\n";
}

int main() {
    omp_init_lock(&lk);
    // Graph: tree with 7 nodes
    adj[0]={1,2}; adj[1]={0,3,4};
    adj[2]={0,5,6}; adj[3]={1};
    adj[4]={1}; adj[5]={2}; adj[6]={2};

    parallel_bfs(0);
    parallel_dfs(0);
    omp_destroy_lock(&lk);
    return 0;
}

#include <bits/stdc++.h>
#include <omp.h>
using namespace std;

void bubble_seq(vector<int>& a) {
    int n = a.size();
    for (int i = 0; i < n-1; i++)
        for (int j = 0; j < n-i-1; j++)
            if (a[j] > a[j+1]) swap(a[j], a[j+1]);
}

// Odd-Even Transposition Sort (parallelizable bubble)
void bubble_par(vector<int>& a) {
    int n = a.size();
    for (int i = 0; i < n; i++) {
        #pragma omp parallel for
        for (int j = i%2; j < n-1; j += 2)
            if (a[j] > a[j+1]) swap(a[j], a[j+1]);
    }
}

void merge(vector<int>& a, int l, int m, int r) {
    vector<int> tmp;
    int i=l, j=m+1;
    while (i<=m && j<=r) tmp.push_back(a[i]<=a[j]?a[i++]:a[j++]);
    while (i<=m) tmp.push_back(a[i++]);
    while (j<=r) tmp.push_back(a[j++]);
    for (int k=l; k<=r; k++) a[k]=tmp[k-l];
}

void merge_par(vector<int>& a, int l, int r, int dep=0) {
    if (l >= r) return;
    int m = (l+r)/2;
    if (dep < 3) {
        #pragma omp parallel sections
        {
            #pragma omp section
            merge_par(a, l, m, dep+1);
            #pragma omp section
            merge_par(a, m+1, r, dep+1);
        }
    } else {
        merge_par(a, l, m, dep+1);
        merge_par(a, m+1, r, dep+1);
    }
    merge(a, l, m, r);
}

int main() {
    srand(42);
    int n = 10000;
    vector<int> a(n), b(n), c(n), d(n);
    for (int i=0; i<n; i++) a[i]=b[i]=c[i]=d[i]=rand()%n;

    double t;

    t = omp_get_wtime();
    bubble_seq(a);
    printf("Seq  Bubble : %.4f s\n", omp_get_wtime()-t);

    t = omp_get_wtime();
    bubble_par(b);
    printf("Par  Bubble : %.4f s\n", omp_get_wtime()-t);

    t = omp_get_wtime();
    sort(c.begin(), c.end());
    printf("Seq  Merge  : %.4f s\n", omp_get_wtime()-t);

    t = omp_get_wtime();
    merge_par(d, 0, n-1);
    printf("Par  Merge  : %.4f s\n", omp_get_wtime()-t);

    return 0;
}

#include <bits/stdc++.h>
#include <omp.h>
using namespace std;

int main() {
    int n = 1000000;
    vector<int> a(n);
    for (int i = 0; i < n; i++) a[i] = rand() % 1000 + 1;

    int    mn  = INT_MAX, mx = INT_MIN;
    long long  sum = 0;

    #pragma omp parallel for \
        reduction(min : mn)  \
        reduction(max : mx)  \
        reduction(+   : sum)
    for (int i = 0; i < n; i++) {
        mn  = min(mn, a[i]);
        mx  = max(mx, a[i]);
        sum += a[i];
    }

    printf("Min : %d\n",  mn);
    printf("Max : %d\n",  mx);
    printf("Sum : %lld\n", sum);
    printf("Avg : %.4f\n", (double)sum / n);
    return 0;
}

#include <iostream>

using namespace std;

__global__ void vecAdd(int* A, int* B, int* C, int n) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < n) C[i] = A[i] + B[i];
}

int main() {
    int n = 1 << 20;            // 1M elements
    int bytes = n * sizeof(int);

    int *hA = new int[n], *hB = new int[n], *hC = new int[n];
    for (int i = 0; i < n; i++) { hA[i] = i; hB[i] = i * 2; }

    int *dA, *dB, *dC;
    cudaMalloc(&dA, bytes); cudaMalloc(&dB, bytes); cudaMalloc(&dC, bytes);
    cudaMemcpy(dA, hA, bytes, cudaMemcpyHostToDevice);
    cudaMemcpy(dB, hB, bytes, cudaMemcpyHostToDevice);

    vecAdd<<<(n+255)/256, 256>>>(dA, dB, dC, n);

    cudaMemcpy(hC, dC, bytes, cudaMemcpyDeviceToHost);
    cout << "C[0]=" << hC[0] << "  C[1]=" << hC[1] << "  C[last]=" << hC[n-1] << "\n";

    cudaFree(dA); cudaFree(dB); cudaFree(dC);
    delete[] hA; delete[] hB; delete[] hC;
    return 0;
}

#define N 512

__global__ void matMul(float* A, float* B, float* C, int n) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    if (row < n && col < n) {
        float s = 0;
        for (int k = 0; k < n; k++) s += A[row*n+k] * B[k*n+col];
        C[row*n+col] = s;
    }
}

int main() {
    int bytes = N*N*sizeof(float);
    float *hA = new float[N*N], *hB = new float[N*N], *hC = new float[N*N];
    for (int i = 0; i < N*N; i++) hA[i] = hB[i] = 1.0f;

    float *dA, *dB, *dC;
    cudaMalloc(&dA,bytes); cudaMalloc(&dB,bytes); cudaMalloc(&dC,bytes);
    cudaMemcpy(dA,hA,bytes,cudaMemcpyHostToDevice);
    cudaMemcpy(dB,hB,bytes,cudaMemcpyHostToDevice);

    dim3 block(16,16), grid((N+15)/16,(N+15)/16);
    matMul<<<grid,block>>>(dA,dB,dC,N);
    cudaDeviceSynchronize();

    cudaMemcpy(hC,dC,bytes,cudaMemcpyDeviceToHost);
    cout << "C[0][0]=" << hC[0] << "  (expected " << N << ")\n";

    cudaFree(dA); cudaFree(dB); cudaFree(dC);
    delete[] hA; delete[] hB; delete[] hC;
    return 0;
}

Parallel forward pass of a 3-layer MLP (784→256→128→10) mimicking MNIST inference with batch processing.

#include <bits/stdc++.h>
#include <omp.h>
using namespace std;

// Parallel matrix multiply: (m x k) * (k x n) = (m x n)
void matmul(float* A, float* B, float* C, int m, int k, int n) {
    #pragma omp parallel for collapse(2)
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++) {
            float s = 0;
            for (int p = 0; p < k; p++) s += A[i*k+p] * B[p*n+j];
            C[i*n+j] = s;
        }
}

void relu(float* x, int n) {
    #pragma omp parallel for
    for (int i = 0; i < n; i++) x[i] = max(0.0f, x[i]);
}

void softmax(float* x, int n) {
    float mx = *max_element(x, x+n), s = 0;
    #pragma omp parallel for reduction(+:s)
    for (int i = 0; i < n; i++) { x[i] = exp(x[i]-mx); s += x[i]; }
    for (int i = 0; i < n; i++) x[i] /= s;
}

int main() {
    // MLP: 784 -> 256 -> 128 -> 10  (MNIST-like)
    int arch[] = {784, 256, 128, 10};
    int L = 3, batch = 32;
    srand(42);
    auto rnd = []{ return (rand()%200-100)/200.0f; };

    vector<vector<float>> W(L), b(L);
    for (int l=0; l<L; l++) {
        W[l].resize(arch[l]*arch[l+1]); for (auto& w:W[l]) w=rnd();
        b[l].resize(arch[l+1]);         for (auto& v:b[l]) v=rnd();
    }

    vector<float> x(batch*arch[0]);
    for (auto& v:x) v=rnd();

    double t = omp_get_wtime();

    vector<float> cur = x;
    for (int l=0; l<L; l++) {
        int in=arch[l], out=arch[l+1];
        vector<float> nxt(batch*out, 0);
        matmul(cur.data(), W[l].data(), nxt.data(), batch, in, out);

        #pragma omp parallel for collapse(2)
        for (int i=0; i<batch; i++)
            for (int j=0; j<out; j++) nxt[i*out+j] += b[l][j];

        if (l < L-1) relu(nxt.data(), batch*out);
        else for (int i=0; i<batch; i++) softmax(nxt.data()+i*out, out);
        cur = nxt;
    }

    printf("Forward pass (%d samples): %.3f ms\n", batch, (omp_get_wtime()-t)*1000);
    printf("Output (sample 0): ");
    for (int i=0; i<10; i++) printf("%.3f ", cur[i]);
    printf("\n");
    return 0;
}
